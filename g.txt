#include "minishell.h"

void	minishell_loop(char *line, char **envp)
{
	t_element	elem;

	elem.cmd = (char *)malloc(20 * sizeof(char));
	elem.type = NULL;
	elem.word = (char *)malloc(20 * sizeof(char));
	elem.type = NULL;
	while ((line = readline("minishell>>")) != NULL)
	{
		add_history(line);
		elem = ft_dispatch(line, elem);
		command_executer(envp, elem);
		free(line);
		line = NULL;
//		rl_replace_line("", 0);
		rl_on_new_line();
		if (strcmp(elem.cmd, "exit") == 0)
			break ;
	}
}

int	main(int argc, char *argv[], char *envp[])
{
	char	*line;
	(void)argc;
	(void)argv;
	line = NULL;
	minishell_loop(line, envp);
}
